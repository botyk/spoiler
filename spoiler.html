Код JS:
<script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
<script>
$(document).ready(function(){
    $("div.spoiler-head", $("body")).unbind("click").click(function(){
		$(this).toggleClass("unfolded");
		$(this).next("div.spoiler-body").slideToggle("fast");
	});
});
</script>
